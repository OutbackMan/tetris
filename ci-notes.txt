.circleci/config.yml:
version: 2
jobs:
  build:
    branches:
      only:
        - master
    macos:
      xcode: "10.0.0"
    steps:
      - checkout
      - restore_cache:
          key: CMAKE_3_12_X
      - run:
          name: Ensuring CMake is Installed
          command: type -P cmake > /dev/null || brew install cmake
      - run:
          name: Configuring
          command: cmake -DCMAKE_C_COMPILER=clang -DCMAKE_BUILD_TYPE=Release
      - save_cache:
          key: CMAKE_3_12_X
          paths: "/usr/local/Cellar/cmake"
